#include<stdio.h> 
int main() 
{ 
int i; 
scanf("%d",&i); 
if(i%2==1 && i%1==0) 
{ 
printf("%d is a prime number.",i); 
 
} 
else 
{ 
printf("%d is not a prime number.",i); 
} 
return 0; 
}
